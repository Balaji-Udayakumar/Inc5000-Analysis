Q5) A Python function with parameters of data frame, category, value, and a number that specifies the number of rows to display in the result.¶

In [246]:
# Function to get the top n companies based on revenues for a specific value in the category

def top_n_companies(df, category=None, value=None, n=10):
    
    # Filtering the DataFrame if both category and value are provided
    if category is not None and value is not None:
        filtered_df = df[df[category] == value]
    # Else assigning back the original dataframe
    else:
        filtered_df = df
    
    # Sorting the filtered DataFrame by revenue in descending order and returning the top n rows
    sorted_df = filtered_df.sort_values(by='REVENUE', ascending=False).head(n)
    
    return sorted_df
